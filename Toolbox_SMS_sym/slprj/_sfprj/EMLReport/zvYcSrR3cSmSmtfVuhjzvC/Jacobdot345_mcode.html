1function [J0dot, Jmdot]=Jacobdot(rp,tp,r0,t0,rL,tL,P0,pm,i,robot)
 2% Computes the geometric Jacobian time-derivative of a point `p`.
 3%
 4% [J0dot, Jmdot]=Jacobdot(rp,tp,r0,t0,rL,tL,P0,pm,i,robot)
 5%
 6% :parameters: 
 7%   * rp -- Position of the point of interest, projected in the inertial CCS -- [3x1].
 8%   * tp -- Twist of the point of interest [\omega,rdot], projected in the intertial CCS -- [6x1].
 9%   * r0 -- Position of the base-link center-of-mass with respect to the origin of the inertial frame, projected in the inertial CCS -- [3x1].
10%   * t0 -- Base-link twist [\omega,rdot], projected in the inertial CCS -- as a [6x1] matrix.
11%   * rL -- Positions of the links, projected in the inertial CCS -- as a [3xn] matrix.
12%   * tL -- Manipulator twist [\omega,rdot], projected in the inertial CCS -- as a [6xn] matrix.
13%   * P0 -- Base-link twist-propagation "vector" -- as a [6x6] matrix.
14%   * pm -- Manipulator twist-propagation "vector" -- as a [6xn] matrix.
15%   * i -- Link id where the point `p` is located -- int 0 to n.
16%   * robot -- Robot model (see :doc:`/Tutorial_Robot`).
17%
18% :return: 
19%   * J0dot -- Base-link Jacobian time-derivative -- as a [6x6] matrix.
20%   * Jmdot -- Manipulator Jacobian time-derivative -- as a [6xn_q] matrix.
21%
22% Examples:
23%
24%   To compute the acceleration of a point ``p`` on the ith link:
25%
26% .. code-block:: matlab
27%   
28%   %Compute Jacobians
29%   [J0, Jm]=Jacob(rp,r0,rL,P0,pm,i,robot);
30%   Compute Jacobians time-derivatives
31%   [J0dot, Jmdot]=Jacobdot(rp,tp,r0,t0,rL,tL,P0,pm,i,robot)
32%   %Twist-rate of that point
33%   tpdot=J0*u0dot+J0dot*u0+Jm*umdot+Jmdot*um;
34%
35% See also: :func:`src.kinematics_dynamics.Accelerations` and :func:`src.kinematics_dynamics.Jacob`. 
36

54%=== CODE ===%
55
56%--- Omega ---%
57%Base-link Omega
58Omega0=[SkewSym(t0(1:3)), zeros(3,3);
59    zeros(3,3), zeros(3,3)];
60
61%Pre-allocate Omega
62Omega=zeros(6,6,robot.n_links_joints,'like',rp);
63
64%Compute Omega
65for j=1:i
66    Omega(1:6,1:6,j)=[SkewSym(tL(1:3,j)), zeros(3,3);
67        zeros(3,3), SkewSym(tL(1:3,j))];
68end
69
70
71%--- Jacobian time-derivative ---%
72
73%Base-link Jacobian
74J0dot=[eye(3),zeros(3,3);SkewSym(r0-rp),eye(3)]*Omega0*P0+[zeros(3,3),zeros(3,3);SkewSym(t0(4:6)-tp(4:6)),zeros(3,3)]*P0;
75
76%Pre-allocate
77Jmdot=zeros(6,robot.n_q,'like',rp);
78    
79%Manipulator Jacobian
80joints_num=0;
81for j=1:i
82    %If joint is not fixed
83    if robot.joints(j).type~=0
84        if robot.con.branch(i,j)==1
85            Jmdot(1:6,robot.joints(j).q_id)=[eye(3),zeros(3,3);SkewSym(rL(1:3,j)-rp),eye(3)]*Omega(1:6,1:6,j)*pm(1:6,j)+[zeros(3,3),zeros(3,3);SkewSym(tL(4:6,j)-tp(4:6)),zeros(3,3)]*pm(1:6,j);
86        else
87            Jmdot(1:6,robot.joints(j).q_id)=zeros(6,1);
88        end
89        joints_num=joints_num+1;
90    end
91end
92
93end
