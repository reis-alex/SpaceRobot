1function [J0, Jm]=Jacob(rp,r0,rL,P0,pm,i,robot)
 2% Computes the geometric Jacobian of a point `p`.
 3%
 4% [J0, Jm]=Jacob(rp,r0,rL,P0,pm,i,robot)
 5% 
 6% :parameters: 
 7%   * rp -- Position of the point of interest, projected in the inertial CCS -- [3x1].
 8%   * r0 -- Position of the base-link, projected in the inertial CCS -- [3x1].
 9%   * rL -- Positions of the links, projected in the inertial CCS -- as a [3xn] matrix.
10%   * P0 -- Base-link twist-propagation "vector" -- as a [6x6] matrix.
11%   * pm -- Manipulator twist-propagation "vector" -- as a [6xn] matrix.
12%   * i -- Link id where the point `p` is located -- int 0 to n.
13%   * robot -- Robot model (see :doc:`/Tutorial_Robot`).
14%
15% :return: 
16%   * J0 -- Base-link geometric Jacobian -- [6x6].
17%   * Jm -- Manipulator geometric Jacobian -- [6xn_q].
18%
19% Examples:
20%
21%   To compute the velocity of the point `p` on the ith link:
22%
23% .. code-block:: matlab
24%   
25%   %Compute Jacobians
26%   [J0, Jm]=Jacob(rp,r0,rL,P0,pm,i,robot);
27%   %Twist of that point
28%   tp=J0*u0+Jm*um;
29%
30% See also: :func:`src.kinematics_dynamics.Kinematics`, :func:`src.kinematics_dynamics.DiffKinematics`
31

49%=== CODE ===%
50
51%--- Number of links  ---%
52
53%Base-link Jacobian
54J0=[eye(3),zeros(3,3);SkewSym(r0-rp),eye(3)]*P0;
55
56%Pre-allocate
57Jm=zeros(6,robot.n_q,'like',rp);
58
59%Manipulator Jacobian
60%Iterate through all "previous" joints
61joints_num=0;
62for j=1:i
63    %If joint is not fixed
64    if robot.joints(j).type~=0
65        if robot.con.branch(i,j)==1
66            Jm(1:6,robot.joints(j).q_id)=[eye(3),zeros(3,3);SkewSym(rL(1:3,j)-rp),eye(3)]*pm(1:6,j);
67        else
68            Jm(1:6,robot.joints(j).q_id)=zeros(6,1);
69        end
70        joints_num=joints_num+1;
71    end
72end
73
74end
