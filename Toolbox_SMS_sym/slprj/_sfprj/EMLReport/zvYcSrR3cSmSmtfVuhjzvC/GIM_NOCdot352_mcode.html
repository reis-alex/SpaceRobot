1function [Hdot]=GIM_NOCdot(N, Ndot,I0,Im,I0_d,Im_d,robot)
 2% Computes the Generalized Inertia Matrix (GIM) Hdot time derivative of the multibody system.
 3%
 4% [Hdot]=GIM_NOCdot(N,Ndot, I0,Im,I0_d, Im_d, robot)
 5% H is unparsed
 6% :parameters: 
 7%   * N -- Natural Orthogonal Complement (NOC) matrix -- a [(6+6*n)x(6+n_q)] matrix.
 8%   * Ndot -- Natural Orthogonal Complesment (NOC) matrix time-derivative -- as a [(6+6*n)x(6+n_q)] matrix.
 9%   * I0 -- Base-link inertia matrix, projected in the inertial CCS -- as a [3x3] matrix.
10%   * Im -- Links inertia matrices, projected in the inertial CCS -- as a [3x3xn] matrix.
11%   * I0_d -- Base-link inertia matrix time derivative, projected in the inertial CCS -- as a [3x3] matrix.
12%   * Im_d -- Links inertia matrices time derivative, projected in the inertial CCS -- as a [3x3xn] matrix.
13%   * robot -- Robot model (see :doc:`/Tutorial_Robot`).
14%
15% :return: 
16%
17%   * Hdot -- Generalized Inertia Matrix time derivative -- as a [(6+n_q)x(6+n_q)] matrix.
18%
19% See also: :func:`src.kinematics_dynamics.GIM_NOCdot`.
20

38%--- generalized mass matrix M ---%
39
40%Pre-allocate M
41M=zeros(6+6*robot.n_links_joints,6+6*robot.n_links_joints,'like',N);
42Mdot=zeros(6+6*robot.n_links_joints,6+6*robot.n_links_joints,'like',N);
43%Base contribution
44M(1:6,1:6)   =[I0(1:3,1:3),zeros(3,3);zeros(3,3),robot.base_link.mass*eye(3)];
45Mdot(1:6,1:6)=[I0_d(1:3,1:3),zeros(3,3);zeros(3,3),zeros(3,3)];
46%Manipulator Contribution
47for i=1:robot.n_links_joints
48    M(6+6*i-5:6+6*i,6+6*i-5:6+6*i)=[Im(1:3,1:3,i),zeros(3,3);zeros(3,3),robot.links(i).mass*eye(3)];
49    Mdot(6+6*i-5:6+6*i,6+6*i-5:6+6*i)=[Im_d(1:3,1:3,i),zeros(3,3);zeros(3,3),zeros(3,3)];
50end
51
52%--- Generalized Inertia Matrix H ---%
53Hdot=N.'*M*Ndot + Ndot.'*M*N + N.'*Mdot*N;
54
55
56end
