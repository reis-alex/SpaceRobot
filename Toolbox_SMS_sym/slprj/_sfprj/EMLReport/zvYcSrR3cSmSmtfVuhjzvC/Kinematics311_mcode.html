1function [RJ,RL,rJ,rL,e,g]=Kinematics(R0,r0,qm,robot)
  2% Computes the kinematics -- positions and orientations -- of the multibody system.
  3%
  4% [RJ,RL,rJ,rL,e,g]=Kinematics(R0,r0,qm,robot)
  5%
  6% :parameters: 
  7%   * R0 -- Rotation matrix from the base-link CCS to the inertial CCS -- [3x3].
  8%   * r0 -- Position of the base-link center-of-mass with respect to the origin of the inertial frame, projected in the inertial CCS -- [3x1].
  9%   * qm -- Displacements of the active joints -- [n_qx1].
 10%   * robot -- Robot model (see :doc:`/Tutorial_Robot`).
 11%
 12% :return: 
 13%   * RJ -- Joints CCS 3x3 rotation matrices with respect to the inertial CCS  -- as a [3x3xn] matrix.
 14%   * RL -- Links CCS 3x3 rotation matrices with respect to the inertial CCS -- as a [3x3xn] matrix.
 15%   * rJ -- Positions of the joints, projected in the inertial CCS -- as a [3xn] matrix.
 16%   * rL -- Positions of the links, projected in the inertial CCS -- as a [3xn] matrix.
 17%   * e -- Joint rotation/sliding axes, projected in the inertial CCS -- as a [3xn] matrix.
 18%   * g -- Vector from the origin of the ith joint CCS to the origin of the ith link CCS, projected in the inertial CCS -- as a [3xn] matrix.
 19%
 20% Remember that all the ouput magnitudes are projected in the **inertial frame**.
 21%
 22% Examples on how to retrieve the results from a specific link/joint:
 23%
 24%   To retrieve the position of the ith link: ``rL(1:3,i)``.
 25%
 26%   To retrieve the rotation matrix of the ith joint: ``RJ(1:3,1:3,i)``.   
 27%
 28% See also: :func:`src.robot_model.urdf2robot` and :func:`src.robot_model.DH_Serial2robot`.
 29

 47%=== CODE ===%
 48
 49%--- Number of links and joints ---%
 50n=robot.n_links_joints;
 51
 52%--- Homogeneous transformation matrices ---%
 53
 54%Pre-allocate homogeneous transformations matrices
 55TJ=zeros(4,4,n,'like',R0);
 56TL=zeros(4,4,n,'like',R0);
 57
 58%--- Base-link ---%
 59T0=[R0,r0;zeros(1,3),1];
 60
 61%--- Forward kinematics recursion ---%
 62
 63%Obtain the joints and links kinematics
 64for i=1:n
 65    
 66    %Get child joint
 67    cjoint=robot.joints(i);
 68    
 69    %Joint kinematics (homogeneous transformation matrix)
 70    if cjoint.parent_link==0
 71        %Parent link is the base-link
 72        TJ(1:4,1:4,cjoint.id)=T0*cjoint.T;
 73    else
 74        %Parent link is not the base-link
 75        TJ(1:4,1:4,cjoint.id)=TL(1:4,1:4,cjoint.parent_link)*cjoint.T;
 76    end
 77    
 78    %Transformation due to current joint variable
 79    if cjoint.type==1
 80        %Revolute
 81        T_qm=[Euler_DCM(cjoint.axis,qm(cjoint.q_id))',zeros(3,1);zeros(1,3),1];
 82    elseif cjoint.type==2
 83        %Prismatic
 84        T_qm=[eye(3),cjoint.axis*qm(cjoint.q_id);zeros(1,3),1];
 85    else
 86        %Fixed
 87        T_qm=[eye(3),zeros(3,1);zeros(1,3),1];
 88    end
 89    
 90    %Link Kinematics (homogeneous transformation matrix)
 91    clink=robot.links(cjoint.child_link);
 92    TL(1:4,1:4,clink.id)=TJ(1:4,1:4,clink.parent_joint)*T_qm*clink.T;
 93end
 94
 95%--- Rotation matrices, translation, position and other geometric quantities ---%
 96
 97%Pre-allocate rotation matrices, translation and positions
 98RJ=zeros(3,3,n,'like',R0);
 99RL=zeros(3,3,n,'like',R0);
100rJ=zeros(3,n,'like',R0);
101rL=zeros(3,n,'like',R0);
102%Pre-allocate rotation/sliding axis
103e=zeros(3,n,'like',R0);
104%Pre-allocate other geometric quantities
105g=zeros(3,n,'like',R0);
106
107%Format rotation matrices, link positions, joint axis and other geometric
108%quantities
109
110%Joint associated quantities
111for i=1:n
112    RJ(1:3,1:3,i)=TJ(1:3,1:3,i);
113    rJ(1:3,i)=TJ(1:3,4,i);
114    e(1:3,i)=RJ(1:3,1:3,i)*robot.joints(i).axis;
115end
116%Link associated quantities
117for i=1:n
118    RL(1:3,1:3,i)=TL(1:3,1:3,i);
119    rL(1:3,i)=TL(1:3,4,i);
120    g(1:3,i)=rL(1:3,i)-rJ(1:3,robot.links(i).parent_joint);
121end
122
123
124end
