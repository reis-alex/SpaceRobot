1function [I0,Im]=I_I(R0,RL,robot)
 2% Projects the link inertias in the inertial CCS.
 3%
 4% [I0,Im]=I_I(R0,RL,robot)
 5%
 6% :parameters: 
 7%   * R0 -- Rotation matrix from the base-link CCS to the inertial CCS -- [3x3].
 8%   * RL -- Links CCS 3x3 rotation matrices with respect to the inertial CCS -- as a [3x3xn] matrix.
 9%   * robot -- Robot model (see :doc:`/Tutorial_Robot`).
10%
11% :return: 
12%   * I0 -- Base-link inertia matrix, projected in the inertial CCS -- as a [3x3] matrix.
13%   * Im -- Links inertia matrices, projected in the inertial CCS -- as a [3x3xn] matrix.
14%
15% See also: :func:`src.kinematics_dynamics.MCB`. 
16

34%=== CODE ===%
35
36%Base-link inertia
37I0 = R0*robot.base_link.inertia*R0';
38
39%Pre-allocate inertias
40Im=zeros(3,3,robot.n_links_joints,'like',R0);
41
42%Inertias of the links
43for i=1:(robot.n_links_joints)
44    Im(1:3,1:3,i)=RL(1:3,1:3,i)*robot.links(i).inertia*RL(1:3,1:3,i)';
45end
46
47end
