1function [t0,tL]=Velocities(Bij,Bi0,P0,pm,u0,um,robot)
 2% Computes the operational-space velocities of the multibody system.
 3%
 4% [t0,tL]=Velocities(Bij,Bi0,P0,pm,u0,um,robot)
 5%
 6% :parameters:
 7%   * Bij -- Twist-propagation matrix (for manipulator i>0 and j>0) -- as a [6x6xn] matrix.
 8%   * Bi0 -- Twist-propagation matrix (for i>0 and j=0) -- as a [6x6xn] matrix.
 9%   * P0 -- Base-link twist-propagation "vector" -- as a [6x6] matrix.
10%   * pm -- Manipulator twist-propagation "vector" -- as a [6xn] matrix.
11%   * u0 -- Base-link velocities [\omega,rdot]. The angular velocity is projected in the body-fixed CCS, while the linear velocity is projected in the inertial CCS -- [6x1].
12%   * um -- Joint velocities -- [n_qx1].
13%   * robot -- Robot model (see :doc:`/Tutorial_Robot`).
14%
15% :return:
16%   * t0 -- Base-link twist [\omega,rdot], projected in the inertial CCS -- as a [6x1] matrix.
17%   * tL -- Manipulator twist [\omega,rdot], projected in the inertial CCS -- as a [6xn] matrix.
18%
19% Use :func:`src.kinematics_dynamics.DiffKinematics` to compute ``Bij``, ``Bi0``, ``P0``, and ``pm``.
20%
21% See also: :func:`src.kinematics_dynamics.Jacob`
22
23

41%=== CODE ===%
42
43%--- Number of links and joints ---%
44n=robot.n_links_joints;
45
46%Pre-allocate
47tL=zeros(6,n,'like',Bij);
48
49%Base-link
50t0=P0*u0;
51
52%Forward recursion to obtain the twist
53for i=1:n
54    
55    if robot.joints(i).parent_link==0
56        %First link
57        tL(1:6,i)=Bi0(1:6,1:6,i)*t0;
58    else
59        %Rest of the links
60        tL(1:6,i)=Bij(1:6,1:6,i,i-1)*tL(1:6,i-1);
61    end
62    
63    %Add joint contribution
64    if robot.joints(i).type~=0
65        tL(1:6,i)=tL(1:6,i)+pm(1:6,i)*um(robot.joints(i).q_id);
66    end
67    
68end
69
70end
